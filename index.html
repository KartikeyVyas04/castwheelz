No premium models located in this catalog view yet.
(Verify your Supabase RLS configuration settings if error persists)
` return } targetGrid.innerHTML = data.map(p => { const activePrice = p.discount_price ? p.discount_price : p.price return `

` : ''}
${p.scale} Class
${p.name}
${p.brand}
